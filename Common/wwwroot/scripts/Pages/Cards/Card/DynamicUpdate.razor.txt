            <CardHeader Title="Robert Jhonson" SubTitle="Chief Executive" />
            @if (IsMedia)
            {
                <CardImage Image="@(SampleService.WebAssetsPath + "images/cards/robert-jhonson.png")" />
            }
            @if (IsText)
            {
                <CardContent Content="Robert received an MBA in marketing from the University of California. He can do marketing for multiple products. He has experience of 10 years in the marketing field." />
            }
            @if (IsFooter)
            {

                    <CardFooterContent Text="Add More" />
                </CardFooter>

        </SfCard>
    </div>
    <div class="col-lg-3 property-section">

        <div class="property-panel-section">
            <div class="property-panel-header">Properties</div>
            <div class="property-panel-content">
                <table class="table">
                    <tr>
                        <td>Card Media</td>
                        <td>
                            <SfCheckBox TChecked="bool" @onchange="@(() => IsMedia = !IsMedia)"></SfCheckBox>
                        </td>
                    </tr>
                    <tr>
                        <td>Card Content</td>
                        <td>
                            <SfCheckBox TChecked="bool" @onchange="@(() => IsText = !IsText)"></SfCheckBox>
                        </td>
                    </tr>
                    <tr>
                        <td>Card Footer</td>
                        <td>
                            <SfCheckBox TChecked="bool" @onchange="@(() => IsFooter = !IsFooter)"></SfCheckBox>
                        </td>
                    </tr>
                </table>
            </div>
        </div>
    </div>
</div>
